		~ y = x - 1
	- else:
		~ y = x + 1
}
---

	(content_block
		(paragraph
			(switch_block

					(if_line (binary (identifier) (number)))
					(then_block
						(code (assignment name: (identifier) value: (binary (identifier) (number)))))
				)
				(switch_arm
					(if_line (else))
					(then_block
						(code (assignment name: (identifier) value: (binary (identifier) (number)))))
				)
			)
		)
	)
)
