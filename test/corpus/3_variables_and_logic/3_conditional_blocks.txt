===
.CCC. Switch Blocks
===
{ x:
- 0: 	zero
- 1: 	one
- 2: 	two
- else: lots
}
---
(ink
	(content_block
		(paragraph
			(cond_block
				(cond_arm (identifier))
				(cond_arm (number) (paragraph (text)))
				(cond_arm (number) (paragraph (text)))
				(cond_arm (number) (paragraph (text)))
				(cond_arm (else) (paragraph (text)))
			)
		)
	)
)
